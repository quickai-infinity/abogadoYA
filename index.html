-- ============================================================ -- ABOGADOYA.APP — Esquema de Base de Datos (Supabase/PostgreSQL) -- ============================================================ CREATE EXTENSION IF NOT EXISTS "uuid-ossp"; CREATE EXTENSION IF NOT EXISTS "postgis"; -- 1. USUARIOS CREATE TABLE public.usuarios ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), auth_id UUID UNIQUE REFERENCES auth.users(id) ON DELETE CASCADE, nombre_completo TEXT NOT NULL, cedula TEXT UNIQUE, telefono TEXT NOT NULL, email TEXT, ciudad TEXT NOT NULL, plan TEXT DEFAULT 'gratis' CHECK (plan IN ('gratis', 'basico', 'premium', 'familiar')), estado TEXT DEFAULT 'activo' CHECK (estado IN ('activo', 'suspendido', 'inactivo')), contacto_emergencia_1_nombre TEXT, contacto_emergencia_1_telefono TEXT, contacto_emergencia_2_nombre TEXT, contacto_emergencia_2_telefono TEXT, contacto_emergencia_3_nombre TEXT, contacto_emergencia_3_telefono TEXT, foto_url TEXT, fecha_registro TIMESTAMPTZ DEFAULT NOW(), ultimo_acceso TIMESTAMPTZ, notificaciones_activas BOOLEAN DEFAULT true, idioma TEXT DEFAULT 'es' ); -- 2. ABOGADOS CREATE TABLE public.abogados ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), auth_id UUID UNIQUE REFERENCES auth.users(id) ON DELETE CASCADE, nombre_completo TEXT NOT NULL, cedula TEXT UNIQUE, telefono TEXT NOT NULL, email TEXT NOT NULL, ciudad TEXT NOT NULL, inpreabogado TEXT NOT NULL, especialidad TEXT NOT NULL, años_experiencia INTEGER DEFAULT 0, universidad TEXT, verificado BOOLEAN DEFAULT false, fecha_verificacion TIMESTAMPTZ, documento_colegiatura_url TEXT, estado TEXT DEFAULT 'pendiente' CHECK (estado IN ('pendiente', 'activo', 'suspendido', 'rechazado')), de_guardia BOOLEAN DEFAULT false, guardia_inicio TIMESTAMPTZ, ubicacion_actual GEOGRAPHY(POINT, 4326), ultima_ubicacion_update TIMESTAMPTZ, calificacion_promedio NUMERIC(3,2) DEFAULT 0.00, total_calificaciones INTEGER DEFAULT 0, total_emergencias_atendidas INTEGER DEFAULT 0, tiempo_respuesta_promedio_seg INTEGER DEFAULT 0, plan_abogado TEXT DEFAULT 'gratuito' CHECK (plan_abogado IN ('gratuito', 'profesional', 'premium')), foto_url TEXT, bio TEXT, fecha_registro TIMESTAMPTZ DEFAULT NOW(), ultimo_acceso TIMESTAMPTZ ); -- 3. EMERGENCIAS CREATE TABLE public.emergencias ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), usuario_id UUID NOT NULL REFERENCES public.usuarios(id), abogado_id UUID REFERENCES public.abogados(id), estado TEXT DEFAULT 'solicitada' CHECK (estado IN ('solicitada', 'buscando_abogado', 'abogado_asignado', 'en_curso', 'finalizada', 'cancelada', 'sin_abogado')), tipo_encuentro TEXT DEFAULT 'alcabala' CHECK (tipo_encuentro IN ('alcabala', 'punto_control', 'detencion', 'requisa', 'transito', 'allanamiento', 'otro')), ubicacion GEOGRAPHY(POINT, 4326) NOT NULL, direccion_aproximada TEXT, ciudad TEXT, estado_venezuela TEXT, fecha_solicitud TIMESTAMPTZ DEFAULT NOW(), fecha_abogado_asignado TIMESTAMPTZ, fecha_inicio_llamada TIMESTAMPTZ, fecha_fin_llamada TIMESTAMPTZ, duracion_llamada_seg INTEGER, tiempo_respuesta_seg INTEGER, sala_video_id TEXT, sala_video_url TEXT, grabacion_url TEXT, grabacion_duracion_seg INTEGER, grabacion_tamaño_mb NUMERIC(10,2), cuerpo_policial TEXT, numero_funcionarios INTEGER, placa_vehiculo_policial TEXT, descripcion_usuario TEXT, notas_abogado TEXT, alertas_contactos_enviadas BOOLEAN DEFAULT false, fecha_alertas_enviadas TIMESTAMPTZ ); -- 4. CALIFICACIONES CREATE TABLE public.calificaciones ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), emergencia_id UUID NOT NULL REFERENCES public.emergencias(id), usuario_id UUID NOT NULL REFERENCES public.usuarios(id), abogado_id UUID NOT NULL REFERENCES public.abogados(id), puntuacion INTEGER NOT NULL CHECK (puntuacion BETWEEN 1 AND 5), comentario TEXT, rapidez INTEGER CHECK (rapidez BETWEEN 1 AND 5), conocimiento INTEGER CHECK (conocimiento BETWEEN 1 AND 5), trato INTEGER CHECK (trato BETWEEN 1 AND 5), fecha TIMESTAMPTZ DEFAULT NOW(), UNIQUE(emergencia_id, usuario_id) ); -- 5. GUARDIAS CREATE TABLE public.guardias ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), abogado_id UUID NOT NULL REFERENCES public.abogados(id), inicio TIMESTAMPTZ NOT NULL, fin TIMESTAMPTZ, activa BOOLEAN DEFAULT true, emergencias_atendidas INTEGER DEFAULT 0, ciudad TEXT, ubicacion_inicio GEOGRAPHY(POINT, 4326) ); -- 6. NOTIFICACIONES CREATE TABLE public.notificaciones ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), usuario_id UUID REFERENCES public.usuarios(id), abogado_id UUID REFERENCES public.abogados(id), emergencia_id UUID REFERENCES public.emergencias(id), tipo TEXT NOT NULL CHECK (tipo IN ('emergencia_nueva', 'abogado_conectado', 'emergencia_finalizada', 'calificacion_pendiente', 'alerta_contacto', 'bienvenida', 'verificacion', 'sistema')), titulo TEXT NOT NULL, mensaje TEXT NOT NULL, leida BOOLEAN DEFAULT false, enviada_push BOOLEAN DEFAULT false, enviada_sms BOOLEAN DEFAULT false, enviada_whatsapp BOOLEAN DEFAULT false, fecha TIMESTAMPTZ DEFAULT NOW() ); -- 7. SUSCRIPCIONES CREATE TABLE public.suscripciones ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), usuario_id UUID REFERENCES public.usuarios(id), abogado_id UUID REFERENCES public.abogados(id), tipo TEXT NOT NULL CHECK (tipo IN ('usuario_basico', 'usuario_premium', 'usuario_familiar', 'abogado_profesional', 'abogado_premium')), precio_mensual_usd NUMERIC(10,2) NOT NULL, estado TEXT DEFAULT 'activa' CHECK (estado IN ('activa', 'cancelada', 'vencida', 'prueba')), fecha_inicio TIMESTAMPTZ DEFAULT NOW(), fecha_proximo_cobro TIMESTAMPTZ, fecha_cancelacion TIMESTAMPTZ, metodo_pago TEXT CHECK (metodo_pago IN ('pago_movil', 'binance', 'zelle', 'paypal', 'stripe', 'otro')), referencia_pago TEXT ); -- 8. PRE-REGISTROS (Landing page) CREATE TABLE public.preregistros ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), tipo TEXT NOT NULL CHECK (tipo IN ('ciudadano', 'abogado')), nombre TEXT NOT NULL, telefono TEXT NOT NULL, ciudad TEXT, email TEXT, inpreabogado TEXT, especialidad TEXT, fecha TIMESTAMPTZ DEFAULT NOW(), convertido BOOLEAN DEFAULT false, fuente TEXT DEFAULT 'landing' ); -- ÍNDICES CREATE INDEX idx_abogados_guardia ON public.abogados(de_guardia, estado, ciudad) WHERE de_guardia = true AND estado = 'activo'; CREATE INDEX idx_abogados_ubicacion ON public.abogados USING GIST(ubicacion_actual); CREATE INDEX idx_emergencias_estado ON public.emergencias(estado) WHERE estado IN ('solicitada', 'buscando_abogado', 'en_curso'); CREATE INDEX idx_emergencias_usuario ON public.emergencias(usuario_id, fecha_solicitud DESC); CREATE INDEX idx_emergencias_abogado ON public.emergencias(abogado_id, fecha_solicitud DESC); CREATE INDEX idx_calificaciones_abogado ON public.calificaciones(abogado_id); CREATE INDEX idx_notificaciones_usuario ON public.notificaciones(usuario_id, leida) WHERE leida = false; CREATE INDEX idx_notificaciones_abogado ON public.notificaciones(abogado_id, leida) WHERE leida = false; CREATE INDEX idx_preregistros_tipo ON public.preregistros(tipo, fecha DESC); -- ROW LEVEL SECURITY ALTER TABLE public.usuarios ENABLE ROW LEVEL SECURITY; ALTER TABLE public.abogados ENABLE ROW LEVEL SECURITY; ALTER TABLE public.emergencias ENABLE ROW LEVEL SECURITY; ALTER TABLE public.calificaciones ENABLE ROW LEVEL SECURITY; ALTER TABLE public.guardias ENABLE ROW LEVEL SECURITY; ALTER TABLE public.notificaciones ENABLE ROW LEVEL SECURITY; ALTER TABLE public.suscripciones ENABLE ROW LEVEL SECURITY; ALTER TABLE public.preregistros ENABLE ROW LEVEL SECURITY; CREATE POLICY "usuarios_own_data" ON public.usuarios FOR ALL USING (auth.uid() = auth_id); CREATE POLICY "abogados_own_data" ON public.abogados FOR ALL USING (auth.uid() = auth_id); CREATE POLICY "abogados_guardia_public" ON public.abogados FOR SELECT USING (de_guardia = true AND estado = 'activo'); CREATE POLICY "emergencias_access" ON public.emergencias FOR ALL USING (usuario_id IN (SELECT id FROM public.usuarios WHERE auth_id = auth.uid()) OR abogado_id IN (SELECT id FROM public.abogados WHERE auth_id = auth.uid())); CREATE POLICY "calificaciones_read" ON public.calificaciones FOR SELECT USING (true); CREATE POLICY "calificaciones_create" ON public.calificaciones FOR INSERT WITH CHECK (usuario_id IN (SELECT id FROM public.usuarios WHERE auth_id = auth.uid())); CREATE POLICY "notificaciones_own" ON public.notificaciones FOR ALL USING (usuario_id IN (SELECT id FROM public.usuarios WHERE auth_id = auth.uid()) OR abogado_id IN (SELECT id FROM public.abogados WHERE auth_id = auth.uid())); CREATE POLICY "preregistros_insert" ON public.preregistros FOR INSERT WITH CHECK (true); -- FUNCIONES CREATE OR REPLACE FUNCTION public.buscar_abogado_guardia(lat DOUBLE PRECISION, lng DOUBLE PRECISION, radio_km INTEGER DEFAULT 50) RETURNS TABLE (abogado_id UUID, nombre TEXT, especialidad TEXT, calificacion NUMERIC, distancia_km DOUBLE PRECISION) LANGUAGE sql STABLE AS $$ SELECT a.id, a.nombre_completo, a.especialidad, a.calificacion_promedio, ST_Distance(a.ubicacion_actual, ST_SetSRID(ST_MakePoint(lng, lat), 4326)::geography) / 1000 FROM public.abogados a WHERE a.de_guardia = true AND a.estado = 'activo' AND a.verificado = true AND ST_DWithin(a.ubicacion_actual, ST_SetSRID(ST_MakePoint(lng, lat), 4326)::geography, radio_km * 1000) ORDER BY 5 ASC LIMIT 5; $$; CREATE OR REPLACE FUNCTION public.actualizar_calificacion_abogado() RETURNS TRIGGER LANGUAGE plpgsql AS $$ BEGIN UPDATE public.abogados SET calificacion_promedio = (SELECT COALESCE(AVG(puntuacion), 0) FROM public.calificaciones WHERE abogado_id = NEW.abogado_id), total_calificaciones = (SELECT COUNT(*) FROM public.calificaciones WHERE abogado_id = NEW.abogado_id) WHERE id = NEW.abogado_id; RETURN NEW; END; $$; CREATE TRIGGER trigger_actualizar_calificacion AFTER INSERT ON public.calificaciones FOR EACH ROW EXECUTE FUNCTION public.actualizar_calificacion_abogado(); CREATE OR REPLACE FUNCTION public.actualizar_emergencias_abogado() RETURNS TRIGGER LANGUAGE plpgsql AS $$ BEGIN IF NEW.estado = 'finalizada' AND NEW.abogado_id IS NOT NULL THEN UPDATE public.abogados SET total_emergencias_atendidas = total_emergencias_atendidas + 1 WHERE id = NEW.abogado_id; END IF; RETURN NEW; END; $$; CREATE TRIGGER trigger_contar_emergencias AFTER UPDATE OF estado ON public.emergencias FOR EACH ROW WHEN (NEW.estado = 'finalizada') EXECUTE FUNCTION public.actualizar_emergencias_abogado();
